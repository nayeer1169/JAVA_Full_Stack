public abstract class Account {
    
    private String accountNumber;
    private String holderName;
    
    protected double balance;

    // Constructor to initialize base details
    public Account(String accountNumber, String holderName, double initialBalance) {
        this.accountNumber = accountNumber; 
        this.holderName = holderName;
        this.balance = initialBalance;
    }

    // Getter - Encapsulated safe read access
    public String getAccountNumber() { return accountNumber; }
    public String getHolderName() { return holderName; }
    public double getBalance() { return balance; }

    // Setter - Encapsulated safe write access with logic checks
    public void setHolderName(String holderName) {
        if (holderName != null && !holderName.trim().isEmpty()) {
            this.holderName = holderName.trim();
        } else {
            System.out.println("Invalid name!");
        }
    }

    public void deposit(double amount) {
        if (amount > 0) {
            this.balance += amount;
            System.out.println("Deposited: " + amount);
        }
    }

    public void withdraw(double amount) {
        if (amount > 0 && this.balance >= amount) {
            this.balance -= amount;
        }
    }

    public abstract void displayDetails();
}
// SavingsAccount inherits all behaviors from Account using 'extends'
public class SavingsAccount extends Account {
    
    private double interestRate;

    public SavingsAccount(String accNum, String name, double bal, double rate) {
        super(accNum, name, bal);
        this.interestRate = rate;
    }

    public void applyInterest() {
        double interest = balance * (interestRate / 100);
        balance += interest;
        System.out.println("Interest applied: " + interest);
    }

    // Implement abstract parent method
    @Override
    public void displayDetails() {
        System.out.println("Savings Account: " + getAccountNumber());
        System.out.println("Holder: " + getHolderName());
        System.out.println("Balance: " + balance);
    }
}
// CurrentAccount inherits from Account
public class CurrentAccount extends Account {
    
    private double overdraftLimit;

    public CurrentAccount(String accNum, String name, double bal, double limit) {
        super(accNum, name, bal);
        this.overdraftLimit = limit;
    }

    // Override withdrawal to support overdraft limits
    @Override
    public void withdraw(double amount) {
        if (amount > 0) {
            if (this.balance + this.overdraftLimit >= amount) {
                this.balance -= amount;
                System.out.println("Withdrew: " + amount);
            } else {
                System.out.println("Overdraft limit exceeded!");
            }
        }
    }

    @Override
    public void displayDetails() {
        System.out.println("Current Account: " + getAccountNumber());
        System.out.println("Holder: " + getHolderName());
        System.out.println("Balance: " + balance);
        System.out.println("Overdraft Limit: " + overdraftLimit);
    }
}
